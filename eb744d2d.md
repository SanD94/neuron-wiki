# Centroid Decomposition (not completed)



```c++
int N;
vector<vector<int>> adj;
vector<int> sze;


int dfs(int s, int e) {
    sze[s] = 1;
    for(auto u : adj[s]) if (u != e) sze[s] += dfs(u, s);
    return sze[s];
}

int centroid(int s, int e) {
    for (auto u : adj[s]) if (u != e) {
        if (sze[u] > N / 2) return centroid(u, s);
    }
    return s;
}

int centroid() {
    dfs(1, 0);
    return centroid(1, 0);
}

void init() {
    cin >> N;
    adj.resize(N+1);
    sze.resize(N+1);
}
```
